f[i][d]代表以 A[i] 结束且公差为 d 的弱等差数列个数
对于所有 j < i，f[i][A[i] - A[j]] += (f[j][A[i] - A[j]] + 1)

```cpp
// 1832 ms
#define ll long long
int numberOfArithmeticSlices(vector<int>& A){
    int n = A.size();
    ll res=0;
    vector<map<ll, int>> cnt(n);
    for(int i=1; i<n; i++){
        for(int j=0; j<i; j++){
            ll delta = (ll)A[i] - (ll)A[j];
            int sum =0;
            if( cnt[j].find(delta)!= cnt[j].end()){
                sum = cnt[j][delta];
            }
            cnt[i][delta] += sum+1;
            res += sum;
        }
    }
    return (int)res;
}
```
